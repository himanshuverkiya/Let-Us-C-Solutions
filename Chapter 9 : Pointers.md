    {
    int x;
    x = *c;
    *c = *b;
    *b = *a;
    *a = x;

    int gcd(int a, int b)
    {
    static int x, temp;
    if (b == 0)
    	return (a);
    x = a / b;
    temp = a;
    a = b;
    b = temp - x*b;
    gcd(a, b);
    }
